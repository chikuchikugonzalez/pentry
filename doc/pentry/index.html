pentry is inspect Process Entry library inspired by mitchellh/go-ps.

Examples

extern crate libc;
extern crate pentry;

let pid: i32;
unsafe {
    pid = libc::getpid() as i32;
}

if let Ok(ps) = pentry::find(pid) {
    println!("#{} {}", ps.pid(), ps.path().unwrap());
}
